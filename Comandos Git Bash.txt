-- COMANDOS UTEIS GIT BASH --

-- ENTRAR EM UMA PASTA DESEJADA -- 

	cd <nome da pasta>

-- SAIR DE ALGUMA PASTA --

	cd ..

-- LIMPAR TELA --

	clear

-- INICIAR REPOSITORIO --

	git init

-- STATUS DO REPOSITORIO --

	git status

-- DEPOIS DE SALVAR OU MODIFICAR UM ARQUIVO NO STAGED AREA, ADICIONA-LO COM O COMANDO: --

	git add <nome&tipo>
	git add *.<tipo> // Adiciona todos no formato selecionado
	git add .        // Adiciona todos os arquivos 

-- DEPOIS COMITAR ELE--

	git commit -m "<versao_de_controle>"
	git commit -a -m "<versao_de_controle>" // Pula a parte anterior e add ao mesmo tempo

-- PARA VER ALTERAÇÕES ANTES DE DAR ADD NA STAGED, USE: --

	git diff		// Em vermelho o que foi deletado, em verde o que foi add
	git diff --staged 	// Caso ja tenha add no staged

-- PARA VER TODOS OS COMMITS--

	git log 	// Mostra apenas usuario e versao que foi alterada
	git log -p  	// Mostra alem do anterior o diff de cada um
	git log --pretty=oneline // Mostra em apenas uma linha cada commit
	gitk		// Mostra um relatorio com interface grafica
