-- COMANDOS UTEIS GIT BASH --

-- ENTRAR EM UMA PASTA DESEJADA -- 

	cd <nome da pasta>

-- SAIR DE ALGUMA PASTA --

	cd ..

-- LIMPAR TELA --

	clear

-- INICIAR REPOSITORIO --

	git init

-- STATUS DO REPOSITORIO --

	git status

-- DEPOIS DE SALVAR OU MODIFICAR UM ARQUIVO NO REPOSITORIO, ADICIONA-LO COM O COMANDO: --

	git add <nome&tipo>
	git add *.<tipo> // Adiciona todos no formato selecionado
	git add .        // Adiciona todos os arquivos 

-- DEPOIS COMITAR ELE

	git commit -m "<versao_de_controle>"
	git commit -a -m "<versao_de_controle>" // Pula a parte anterior e add ao mesmo tempo
